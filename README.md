## Description

[Nest](https://github.com/nestjs/nest) framework TypeScript starter repository.

## Installation

```bash
$ npm install
```

## Running the app

```bash
# development
$ npm run start

# watch mode
$ npm run start:dev

# production mode
$ npm run start:prod
```

## Test

```bash
# unit tests
$ npm run test

# e2e tests
$ npm run test:e2e

# test coverage
$ npm run test:cov
```

# nest.js란
# node를 확장하여 typescript 기반으로 api 제작과정에서 oop, di, singletom 등을 적용한 프레임워크

# Module
# 모듈은 api 하나당 적어도 하나의 module이 존재하고 각 객체들이 자동으로 주입될 수 있도록 어플리케이션 구조를 받혀주는 요소
# nest.js가 싱글톤 패턴을 지향하기 때문에 인스턴스를 직접 생성하지 않고 모듈을 통해 injection 하는 패턴을 권장

# 기본적인 형태

import { Module } from '@nestjs/common';
import { CatsController } from './cats.controller';
import { CatsService } from './cats.service';

@Module({
  imports: [DogModule], # 다른 모듈에서 export한 모듈을 현재 모듈에서 사용할 목록
  controllers: [CatsController], # 해당 모듈에서 사용할 컨트롤러
  providers: [CatsService, ExternalService], # 해당 모듈에서 사용하며 인스턴스 화 시킬 서비스
  exports: [CatsService, DogModule] # 해당 모듈에서 제공하는 providers 중 외부와 다른 모듈에서 사용할 수 있도록 할 서비스
})

export class CatsModule{}
